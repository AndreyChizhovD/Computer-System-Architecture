### Код на языке C
```c
#include <stdio.h>

static int A[1048576];
static int B[1048576];

int main(int argc, char** argv) {
    int n;
    int i;

    scanf("%d", &n);
    for (i = 0; i < n; ++i) {
        scanf("%d", &A[i]);
        
    }

    for (i = 0; i < n; ++i) {
    	B[i] = A[i] * A[i];
    }

    for (i = 0; i < n; ++i) {
    	printf("%d", B[i]);
    }

    return 0;
}
```
